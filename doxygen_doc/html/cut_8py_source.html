00047 #     Execute a graph cut using the supplied source file of Boyov and Kolmogorovs
00048 #     max-flow/min-cut algorithm.
00049 #     
00050 #     The source-file is written to a temporary directory, compiled and the executed. The
00051 #     created output is caught and returned.
00052 #     
00053 #     @note: This function assumes g++ - GNU project C++ compiler to be installed
00054 #     on the host machine. Alternatively the path to g++ can be provided.
00055 #     
00056 #     @warning: If not handled with care, this function can be used to compile and execute
00057 #     arbitrary code on the host machine. Ensure that the input of this function can not
00058 #     be generated or modified by the user and always comes from @link: bk_mfmc_generate().
00059 #     
00060 #     @param source_file: The C++ source file as returned by @link: bk_mfmc_generate().
00061 #     @type source_file: str
00062 #     @param gpp_location: The location of g++ - the GNU project C++ compiler.
00063 #     @type gpp_location: str
00064 #     @param keep: Set this to true to keep the temporary directory after execution.
00065 #                  This parameter is intended for debugging purposes only.
00066 #     @type keep: bool
00067 #     @return: The results as a string to be parsed by @link: bk_mfmc_parse().
00068 #     @rtype: str
00069 #     
00070 #     @raise SubprocessError: When the one step of the compiling or execution can not be
00071 #                             assured to have terminated succesfully.
00072 #     
00073 def bk_mfmc_cut(source_file, gpp_location = False, keep = False):
00074     # prepare logger
00075     logger = Logger.getInstance()
00076     
00077     logger.info('Creating temporary file structure...')
00078     
00079     # get the algorithms library
00080     library = __bk_mfmc_get_library()
00081     
00082     # prepare temporary file structure and library links
00083     try:
00084         tmp_dir = tempfile.mkdtemp()
00085     except Exception as e:
00086         raise SubprocessError('Failed to reate a temporary directory, reason {}.'.format(e.message), e)
00087     executable = tmp_dir + '/compute'
00088     mygraph_file = tmp_dir + '/mygraph.cpp'
00089     
00090     logger.debug('Temporary files can be found in {} (set keep=True to keep)'.format(tmp_dir))
00091 
00092     # generate cpp file with the graph encoded
00093     with open(mygraph_file, 'w') as f:
00094         f.write(source_file)
00095     
00096     # compile the application
00097     logger.info('Compiling...')
00098     if not gpp_location: gpp_location = 'g++'
00099     cmd = [gpp_location, '-o', executable, mygraph_file, library['graph_file'], library['maxflow_file']]
00100     logger.debug('Command: ' + ' '.join(cmd))
00101     try:
00102         fnull = open(os.devnull, 'w')
00103         subprocess.check_call(cmd, stdout=fnull, stderr=fnull)
00104     except subprocess.CalledProcessError as e:
00105         if not keep: __bk_mfmc_clean_tmp(tmp_dir, mygraph_file)
00106         raise SubprocessError('Compiling the application failed with exit code {} and message {}.'.format(e.returncode, e.output))
00107         
00108     # execute the binary and collect the output
00109     logger.info('Executing...')
00110     try:
00111         result = subprocess.check_output(executable)
00112     except subprocess.CalledProcessError as e:
00113         if not keep: __bk_mfmc_clean_tmp(tmp_dir, mygraph_file, executable)
00114         raise SubprocessError('Executing the binary returned an exit code {} with message {} ({}).'.format(e.returncode, e.output, e.message))
00115     except Exception as e:
00116         if not keep: __bk_mfmc_clean_tmp(tmp_dir, mygraph_file, executable)
00117         raise SubprocessError('Executing the binary failed due to reason {}.'.format(e.message))    
00118     
00119     if not keep:
00120         try:
00121             os.remove(mygraph_file)
00122             os.remove(executable)
00123             os.rmdir(tmp_dir)
00124         except Exception as e:
00125             warnings.warn('Could not clean up all of the temporary files, reason {}. They can be found in {} for manually cleaning, but that should not be necessary.'.format(e.message, tmp_dir))
00126     
00127     return result
