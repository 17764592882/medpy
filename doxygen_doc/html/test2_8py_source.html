00101 def gauss(x, y, sigma):
00102     return math.exp(-1 * (math.pow(x, 2) + math.pow(x, 2)) / 2 * math.pi * math.pow(sigma, 2))

00104 def laplace(x, y, sigma):
00105     a = -1. / (math.pi * math.pow(sigma, 4))
00106     b = 1 - (math.pow(x, 2) + math.pow(y, 2)) / (2 * math.pow(sigma, 2))
00107     return a * b * gauss(x, y, sigma)

00109 def LoG2d(x, y, sigma):
00110     xysqr = math.pow(x, 2) + math.pow(y, 2)
00111     ssqrdbl = -2 * math.pow(sigma, 2)
00112     term = xysqr/ssqrdbl
00113     return -1. / (math.pi * math.pow(sigma, 4)) * (1 + term) * math.exp(term)

00121 def features(image): # 100^3 in 60-100ms
00122     mean = image.mean()
00123     img = image - image.min() # remove eventual negative values
00124     dist = scipy.bincount(img.ravel())/float(img.size) # compute probability histogram (work only for ints)
00125     entropy = 0
00126     uniformity = 0
00127     for grey_value in scipy.unique(img):
00128         entropy += dist[grey_value] * math.log(dist[grey_value], 2)
00129         uniformity += math.pow(dist[grey_value], 2)
00130     entropy *= -1
00131     return mean, entropy, uniformity
