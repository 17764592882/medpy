00039 # 
00040 #     Apply a region-to-source/sink mapping to a label image.
00041 #     
00042 #     Takes the original label image and a mapping from its region ids to source/sink as
00043 #     returned by one of this modules *_parse functions. From this a binary mask is created
00044 #     where True values denote source and False sink affiliation for each voxel.
00045 #     
00046 #     @note if the provided label_image is a ndarray, the operation can be executed on the
00047 #     array directly, changing its values. If you do not want this to happen, provide a
00048 #     copy instead of the original array.
00049 #     
00050 #     @param label_image: A nD label_image
00051 #     @type label_image: sequence
00052 #     @param mapping: A mapping of the label_images region ids to source or sink.
00053 #     @type mapping: dict
00054 #     
00055 #     @return: A binary image
00056 #     @rtype: numpy.ndarray
00057 #     
00058 #     @raise ArgumentError: If a region id is missing in the supplied mapping
00059 #     
00060 def apply_mapping(label_image, mapping):
00061     label_image = scipy.array(label_image)
00062     rav = label_image.ravel()
00063     for i in range(len(rav)):
00064         if not rav[i] in mapping:
00065             raise ArgumentError('No conversion for region id {} found in the supplied mapping.'.format(rav[i]))
00066         rav[i] = mapping[rav[i]]
00067     return rav.reshape(label_image.shape).astype(scipy.bool_)

00076 #     Parse the output generated by a Boyov and Kolmogorovs max-flow/min-cut algorithm
00077 #     execution.
00078 #     
00079 #     The result maps from the region-ids of the original image to either source or sink.
00080 #     
00081 #     The original source file has to be generated using @link: bk_mfmc_generate() to
00082 #     ensure correct parsing.
00083 #     
00084 #     @param output: The output returned by @link: bk_mfmc_cut() or a manual execution.
00085 #     @type output: str
00086 #     @return: (maxflow, map) where maxflow is and int and map a dictionary containing
00087 #              node_id->BK_MFMC_SOURCE_MARKER / BK_MFMC_SINK_MARKER ({node_id: source|sink, ...})
00088 #              mappings (@link: graph.BK_MFMC_SOURCE_MARKER, @link: graph.BK_MFMC_SINK_MARKER).
00089 #     @rtype: dict
00090 #     
00091 #     @raise ArgumentError: When the supplied output is maleformed.
00092 #     
00093 def bk_mfmc_parse(output):
00094     # prepare logger
00095     logger = Logger.getInstance()
00096     
00097     logger.info('Parsing results...')
00098     try:
00099         lines = output.split('\n')
00100         flow = int(lines[0][lines[0].find('=') + 1:])
00101         nodes = {}
00102         for line in lines[1:]:
00103             line = line.strip()
00104             if 0 == len(line) or '#' == line[0]: continue # skip comments and empty lines
00105             nodes[int(line[:line.find('=')])] = int(line[line.find('=') + 1:])
00106         logger.debug('flow={}, #nodes={}, #source={}, #sink={}'.format(flow, 
00107                                                                        len(nodes), 
00108                                                                        sum([1 if x == __BK_MFMC_SOURCE_MARKER else 0 for x in nodes.values()]),
00109                                                                        sum([1 if x == __BK_MFMC_SINK_MARKER else 0 for x in nodes.values()])))
00110         return (flow, nodes)
00111     except ValueError as e:
00112         raise ArgumentError('Maleformed results argument: {}.'.format(e.message), e)
