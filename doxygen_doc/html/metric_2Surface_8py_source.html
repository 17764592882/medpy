00026 # 
00027 #     Computes different surface metrics between two 3D-images contain each an object.
00028 #     The surface of the objects is computed using a 18-neighbourhood edge detection.
00029 #     The distance metrics are computed over all points of the surfaces using the nearest
00030 #     neighbour approach.
00031 #     Beside this provides a number of statistics of the two images.
00032 #     
00033 #     During the initialization the edge detection is run for both images, taking up to
00034 #     5 min (on 512^3 images). The first call to one of the metric measures triggers the
00035 #     computation of the nearest neighbours, taking up to 7 minutes (based on 250.000 edge
00036 #     point for each of the objects, which corresponds to a typical liver mask). All
00037 #     subsequent calls to one of the metrics measures can be expected be in the
00038 #     sub-millisecond area.
00039 #     
00040 #     Metrics defined in:
00041 #     Heimann, T.; van Ginneken, B.; Styner, M.A.; Arzhaeva, Y.; Aurich, V.; Bauer, C.; Beck, A.; Becker, C.; Beichel, R.; Bekes, G.; Bello, F.; Binnig, G.; Bischof, H.; Bornik, A.; Cashman, P.; Ying Chi; Cordova, A.; Dawant, B.M.; Fidrich, M.; Furst, J.D.; Furukawa, D.; Grenacher, L.; Hornegger, J.; Kainmuller, D.; Kitney, R.I.; Kobatake, H.; Lamecker, H.; Lange, T.; Jeongjin Lee; Lennon, B.; Rui Li; Senhu Li; Meinzer, H.-P.; Nemeth, G.; Raicu, D.S.; Rau, A.-M.; van Rikxoort, E.M.; Rousson, M.; Rusko, L.; Saddi, K.A.; Schmidt, G.; Seghers, D.; Shimizu, A.; Slagmolen, P.; Sorantin, E.; Soza, G.; Susomboon, R.; Waite, J.M.; Wimmer, A.; Wolf, I.; , "Comparison and Evaluation of Methods for Liver Segmentation From CT Datasets," Medical Imaging, IEEE Transactions on , vol.28, no.8, pp.1251-1265, Aug. 2009

00107     ##
00108     # 
00109     #         Computes the maximum symmetric surface distance, also known as Hausdorff
00110     #         distance, between the two objects surfaces.
00111     #         
00112     #         @return: the maximum symmetric surface distance in millimeters
00113     #         
00114     #         For a perfect segmentation this distance is 0. This metric is sensitive to
00115     #         outliers and returns the true maximum error.
00116     #         
00117     #         Metric definition:
00118     #         Let \f$S(A)\f$ denote the set of surface voxels of \f$A\f$. The shortest
00119     #         distance of an arbitrary voxel \f$v\f$ to \f$S(A)\f$ is defined as:
00120     #         \f[
00121     #             d(v,S(A)) = \min_{s_A\in S(A)} ||v-s_A||
00122     #         \f]
00123     #         where \f$||.||\f$ denotes the Euclidean distance. The maximum symmetric
00124     #         surface distance is then given by:
00125     #         \f[
00126     #             MSD(A,B) = \max
00127     #                 \left\{
00128     #                     \max_{s_A\in S(A)} d(s_A,S(B)),
00129     #                     \max_{s_B\in S(B)} d(s_B,S(A)),
00130     #                 \right\}
00131     #         \f]
00132     #         
00133     def GetMaximumSymmetricSurfaceDistance(self):
00134         # Get the maximum of the nearest neighbour distances
00135         A_B_distance = self.GetMaskReferenceNn().max()
00136         B_A_distance = self.GetReferenceMaskNn().max()
00137         
00138         # compute result and return
00139         return max(A_B_distance, B_A_distance)

00141     ##
00142     # 
00143     #         Computes the root mean square symmetric surface distance between the
00144     #         two objects surfaces.
00145     #         
00146     #         @return: root mean square symmetric surface distance in millimeters
00147     #         
00148     #         For a perfect segmentation this distance is 0. This metric punishes large
00149     #         deviations from the true contour stronger than the average symmetric surface
00150     #         distance.
00151     #         
00152     #         Metric definition:
00153     #         Let \f$S(A)\f$ denote the set of surface voxels of \f$A\f$. The shortest
00154     #         distance of an arbitrary voxel \f$v\f$ to \f$S(A)\f$ is defined as:
00155     #         \f[
00156     #             d(v,S(A)) = \min_{s_A\in S(A)} ||v-s_A||
00157     #         \f]
00158     #         where \f$||.||\f$ denotes the Euclidean distance. The root mean square
00159     #         symmetric surface distance is then given by:
00160     #         \f[
00161     #           RMSD(A,B) = 
00162     #             \sqrt{\frac{1}{|S(A)|+|S(B)|}}
00163     #             \times
00164     #             \sqrt{
00165     #                 \sum_{s_A\in S(A)} d^2(s_A,S(B))
00166     #                 +
00167     #                 \sum_{s_B\in S(B)} d^2(s_B,S(A))
00168     #             }
00169     #         \f]
00170     #         
00171     def GetRootMeanSquareSymmetricSurfaceDistance(self):
00172         # get object sizes
00173         mask_surface_size = len(self.GetMaskEdgePoints())
00174         reference_surface_sice = len(self.GetReferenceEdgePoints())
00175         
00176         # get minimal nearest neighbours distances
00177         A_B_distances = self.GetMaskReferenceNn()
00178         B_A_distances = self.GetReferenceMaskNn()
00179         
00180         # square the distances
00181         A_B_distances_sqrt = A_B_distances * A_B_distances
00182         B_A_distances_sqrt = B_A_distances * B_A_distances
00183         
00184         # sum the minimal distances
00185         A_B_distances_sum = A_B_distances_sqrt.sum()
00186         B_A_distances_sum = B_A_distances_sqrt.sum()
00187 
00188         # compute result and return
00189         return math.sqrt(1. / (mask_surface_size + reference_surface_sice)) * math.sqrt(A_B_distances_sum + B_A_distances_sum)

00195     #         
00196     #         @return: average symmetric surface distance in millimeters

00198     #         For a perfect segmentation this distance is 0.
00199     #         
00200     #         Metric definition:
00201     #         Let \f$S(A)\f$ denote the set of surface voxels of \f$A\f$. The shortest
00202     #         distance of an arbitrary voxel \f$v\f$ to \f$S(A)\f$ is defined as:
00203     #         \f[
00204     #             d(v,S(A)) = \min_{s_A\in S(A)} ||v-s_A||
00205     #         \f]
00206     #         where \f$||.||\f$ denotes the Euclidean distance. The average symmetric
00207     #         surface distance is then given by:
00208     #         \f[
00209     #             ASD(A,B) = 
00210     #                 \frac{1}{|S(A)|+|S(B)|}
00211     #                 \left(
00212     #                     \sum_{s_A\in S(A)} d(s_A,S(B))
00213     #                     +
00214     #                     \sum_{s_B\in S(B)} d(s_B,S(A))
00215     #                 \right)
00216     #         \f]
00217     #         
00218     def GetAverageSymmetricSurfaceDistance(self):
00219         # get object sizes
00220         mask_surface_size = len(self.GetMaskEdgePoints())
00221         reference_surface_sice = len(self.GetReferenceEdgePoints())
00222         
00223         # get minimal nearest neighbours distances
00224         A_B_distances = self.GetMaskReferenceNn()
00225         B_A_distances = self.GetReferenceMaskNn()

00227         # sum the minimal distances
00228         A_B_distances = A_B_distances.sum()

00246     ##
00247     # 
00248     #         @return: The distances of the nearest neighbours of all reference edge points
00249     #                  to all mask edge points.
00250     #                  
00251     #         The underlying algorithm used for the scipy.spatial.KDTree implementation is
00252     #         based on:
00253     #         Sunil Arya, David M. Mount, Nathan S. Netanyahu, Ruth Silverman, and
00254     #         Angela Y. Wu. 1998. An optimal algorithm for approximate nearest neighbor
00255     #         searching fixed dimensions. J. ACM 45, 6 (November 1998), 891-923

00257     def GetReferenceMaskNn(self):
00258         # Note: KDTree is faster than scipy.spatial.distance.cdist when the number of
00259         # voxels exceeds 10.000 (computationally tested). The maximum complexity is
00260         # O(D*N^2) vs. O(D*N*log(N), where D=3 and N=number of voxels
00261         if None == self._reference_mask_nn:
00262             tree = scipy.spatial.cKDTree(self.GetReferenceEdgePoints())
00263             self._reference_mask_nn, _ = tree.query(self.GetMaskEdgePoints())
00264         return self._reference_mask_nn

00283     #         Uses a 18-neighbourhood filter to create an edge image of the input object.
00284     #         Assumes the input to be a representation of a 3D image, that fits one of the
00285     #         following formats:
00286     #             1. all 0 values denoting background, all others the foreground/object
00287     #             2. all False values denoting the background, all others the foreground/object
00288     #         The area outside the array is assumed to contain background voxels. The method
00289     #         does not ensure that the object voxels are actually connected, this is silently
00290     #         assumed.
00291     #         
00292     #         @param array: a numpy array with only 0/N\{0} or False/True values.
00293     #         @return: a boolean numpy array with the input objects edges
00294     #         
00295     def ComputeContour(array):
00296         # set 18-neighbourhood/conectivity (for 3D images) alias face-and-edge kernel
00297         # all values covered by 1/True passed to the function 
00298         # as a 1D array in order left-right, top-down
00299         # Note: all in all 19 ones, as the center value
00300         # also has to be checked (if it is a masked pixel)
00301         # [[[0, 1, 0], [[1, 1, 1],  [[0, 1, 0], 
00302         #   [1, 1, 1],  [1, 1, 1],   [1, 1, 1],
00303         #   [0, 1, 0]], [1, 1, 1]],  [0, 1, 0]]]
00304         footprint = scipy.ndimage.morphology.generate_binary_structure(3, 2)
00305         
00306         # create an erode version of the array
00307         erode_array = scipy.ndimage.morphology.binary_erosion(array, footprint)
00308         
00309         # xor the erode_array with the original and return
00310         return array ^ erode_array
