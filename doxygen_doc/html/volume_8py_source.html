00073     # 
00074     #         Returns the relative volume difference between two shapes defined by the
00075     #         non-background voxels of in the binary images.
00076     #         
00077     #         The relative volume distance of 0 means that the volumes are identical. Note that
00078     #         this doesn't apply that the shapes are identical or overlap at all. The result is
00079     #         given as signed number [-100,+100] to indicate over- or under-segmentation.
00080     #         Warning: Can produce values greater than -/+100 if the difference in volume
00081     #         exceeds 50%. Positive values means that the mask volume is too big, a negative
00082     #         value that it is too small.
00083     #         
00084     #         Warning: This measure is not actually a metric, as it is not symmetric.
00085     #         
00086     #         Metric definition:
00087     #         The relative volume difference between two sets of voxels \f$S(A)\f$ and \f$S(b)\f$
00088     #         is given in percent and defined as:
00089     #         \f[
00090     #             100 * \frac{|A - B|}{|B|}
00091     #         \f]        
00092     #         
00093     def GetRelativeVolumeDifference(self):
00094         # compute the objects sizes
00095         mask_size = self.GetMaskSize()
00096         reference_size = self.GetReferenceSize()
00097         
00098         # compute return value
00099         return 100. * float(mask_size - reference_size) / reference_size

00174     def _ComputeIntersectionWindow(self):
00175         mask_rec = scipy.array([self._mask_offset, # lower-left corner
00176                                 self._mask_offset + scipy.array(self._mask.shape)]) # upper-right corner
00177         reference_rec = scipy.array([self._reference_offset, # lower-left corner
00178                                       self._reference_offset + scipy.array(self._reference.shape)]) # upper-right corner
00179         
00180         # check if an intersection occurs at all
00181         intersect = mask_rec[0][0] < reference_rec[1][0] and \
00182                     mask_rec[0][1] < reference_rec[1][1] and \
00183                     mask_rec[0][2] < reference_rec[1][2] and \
00184                     mask_rec[1][0] > reference_rec[0][0] and \
00185                     mask_rec[1][1] > reference_rec[0][1] and \
00186                     mask_rec[1][2] > reference_rec[0][2]
00187                     
00188         # if not, set an empty intersection
00189         if not intersect:
00190             self._intersection_window = scipy.array([[0,0,0],[0,0,0]]) # cube, offset
00191             return
00192         
00193         # otherwise compute intersection cube
00194         intersection = ((max(mask_rec[0][0], reference_rec[0][0]), #x lower
00195                          max(mask_rec[0][1], reference_rec[0][1]), #y lower
00196                          max(mask_rec[0][2], reference_rec[0][2])), #z lower
00197                         (min(mask_rec[1][0], reference_rec[1][0]), #x upper
00198                          min(mask_rec[1][1], reference_rec[1][1]), #y upper
00199                          min(mask_rec[1][2], reference_rec[1][2]))) #z upper
00200         
00201         # set intersection size (cube)
00202         self._intersection_window = scipy.array(intersection)

00216         w_m = self._mask[iw_m[0][0]:iw_m[1][0], iw_m[0][1]:iw_m[1][1], iw_m[0][2]:iw_m[1][2]]
