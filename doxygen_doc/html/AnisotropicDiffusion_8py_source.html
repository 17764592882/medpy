00025     # by: Zachary Pincus
00026     # Anisotropic Diffusion, as per Perona and Malik's paper (see section V).

00032     def _exp(self, image_gradient, scale):
00033         return numpy.exp(-(numpy.absolute(image_gradient)/scale)**2)

00036     def _inv(self, image_gradient, scale):

00039     def anisotropic_diffusion(self, image, num_iters=10, scale=10, step_size=0.2, conduction_function=AnisotropicDiffusion._inv):
00040         # 'step_size' is Perona and Malik's lambda parameter; scale is their 'K' parameter.
00041         # The 'conduction_function' is the function 'g' in the original formulation;
00042         # if this function simply returns a constant, the result is Gaussian blurring.
00043         if step_size > 0.25:
00044             raise ValueError('step_size parameter must be <= 0.25 for numerical stability.')
00045         image = image.copy()
00046         # simplistic boundary conditions -- no diffusion at the boundary
00047         central = image[1:-1, 1:-1]
00048         n = image[:-2, 1:-1]
00049         s = image[2:, 1:-1]
00050         e = image[1:-1, :-2]
00051         w = image[1:-1, 2:]
00052         directions = [s,e,w]
00053         for i in range(num_iters):
00054             di = n - central
00055             accumulator = conduction_function(di, scale)*di
00056             for direction in directions:
00057                 di = direction - central
00058                 accumulator += conduction_function(di, scale)*di
00059             accumulator *= step_size
00060             central += accumulator
00061         return image
