00051 #     In the return tuple, the 'nodes' list holds the region-ids of all regions in the

00053 #     with (region-id-a, region-id-b) tuples as keys and the corresponding weights as values;
00054 #     and 't_weights' a dictionary with the region-ids as keys and a tuple like
00055 #     (weight_to_source, weight_to_sink) as values. When the 'directed' parameter was set
00056 #     to True, 'n_weights' contains also entries for (region-id-b, region-id-a).
00057 #     The neighbourhood used is the 6-connectivity as returned by
00058 #     generate_binary_structure(3, 1).
00059 #     
00060 #     @param image: The original image as an array of arbitrary type.
00061 #     @param label_image: The label image as an array containing uint values.

00063 #     @param bg_markers: The background markers as binary array.
00064 #     @param regional_term: This can be either
00065 #                           False - 
00066 #                           No weights are computed for the t_edges leading from
00067 #                           source to the node and from the node to the sink.
00068 #                           Instead all are set to 0, except for the nodes that are
00069 #                           directly connected to the source or sink, as determined
00070 #                           by the fg and bg markers, which are set to +inf.

00073 #                           have the following signature

00080 #                           an array holding the region the current node corresponds to;

00082 #                           via the regional_term_args argument.
00083 #     @param boundary_term: This can be either

00089 #                           supplied function has to have the following signature
00090 #                           weight = fun(region-a, region-b [,boundary_term_args])
00091 #                           and is called for any two neighbouring regions. If the directed
00092 #                           parameter is set to True, it is also called for the computation

00367 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00368 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00369 #     supplied region/label image.
00370 #     Note: The returned set contains neither duplicates, nor self-references
00371 #     (i.e. (id_1, id_1)), nor reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00372 #     
00373 #     @param label_image: An image with labeled regions (nD).
00374 #     @param return: A set with tuples denoting the edge neighbourhood.
00375 #     
00376 def __compute_edges(label_image):
00377     # compute the neighbours
00378     Er = __compute_edges_nd(label_image)
00379     
00380     # remove reversed neighbours and self-references
00381     for edge in list(Er):
00382         if (edge[1], edge[0]) in Er:
00383             Er.remove(edge)
00384     return Er

00388 # 
00389 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00390 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00391 #     supplied region/label image.
00392 #     Note: The returned set can contain self references (i.e. (id_1, id_1)) as well as
00393 #     reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00394 #     @see: __compute_edges_nd2 alternative implementation (slighty slower)
00395 #     @see: __compute_edges_3d faster implementation for three dimensions only
00396 #     
00397 #     @param label_image: An image with labeled regions (nD).
00398 #     @param return: A set with tuples denoting the edge neighbourhood.
00399 #     
00400 def __compute_edges_nd(label_image):
00401     Er = set()
00402     for dim in range(label_image.ndim):
00403         slices_x = []
00404         slices_y = []
00405         for di in range(label_image.ndim):
00406             slices_x.append(slice(None, -1 if di == dim else None))
00407             slices_y.append(slice(1 if di == dim else None, None))
00408         Er = Er.union(set(zip(label_image[slices_x].flat,
00409                               label_image[slices_y].flat)))
00410     return Er

00412 ##
00413 # 
00414 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00415 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00416 #     supplied region/label image.
00417 #     Note: The returned set can contain self references (i.e. (id_1, id_1)) as well as
00418 #     reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00419 #     @see: __compute_edges_nd alternative implementation (slightly faster)
00420 #     @see: __compute_edges_3d faster implementation for three dimensions only
00421 #     
00422 #     @param label_image: An image with labeled regions (nD).
00423 #     @param return: A set with tuples denoting the edge neighbourhood.
00424 #     
00425 def __compute_edges_nd2(label_image):
00426     # prepare iterators
00427     labels = label_image.flat
00428     nb_pointers = []
00429     for dim in range(label_image.ndim):
00430         slices_base = []
00431         slices_add = []
00432         for di in range(label_image.ndim):
00433             slices_base.append(slice(1 if di == dim else None, None))
00434             slices_add.append(slice(-1 if di == dim else None, None))
00435         nb_pointers.append(scipy.concatenate((label_image[slices_base],
00436                                               label_image[slices_add]), dim).flat)
00437     
00438     # iterate and search neighbours
00439     Er = set()
00440     for label, nbs in zip(labels, zip(*nb_pointers)):
00441         for nb in nbs: Er.add((label, nb))
00442     return Er
