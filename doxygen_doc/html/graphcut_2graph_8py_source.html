00042 #     Represents a graph suitable for further processing with the graphcut package.
00043 #     
00044 #     The graph contains nodes, edges (directed) between the nodes (n-edges), edges
00045 #     between two terminals (called source and sink) and the edges (t-edges) and a
00046 #     weight for each edge. 
00047 #     
00048 #     Methods:
00049 #     set_nodes(nodes) -- Set the nodes.
00050 #     set_source_nodes(source_nodes) -- Set the source nodes.
00051 #     set_sink_nodes(sink_nodes) -- Set the sink nodes.
00052 #     set_nweights(nweights) -- Set the inter-node weights-
00053 #     add_tweights(nweights) -- Add some terminal-to-node weights.
00054 #     get_nodes() -- Return all nodes.
00055 #     get_source_nodes() -- Return all nodes connected to the source.
00056 #     get_sink_nodes() -- Return all nodes connected to the sink.
00057 #     get_edges() -- Return all edges.
00058 #     get_nweights() -- Return all inter-node weights.

00238 #     Create a @link: Graph object from a nD label image.
00239 #     
00240 #     The graph created consists of nodes, edges and weights (two type: n-weight, which are
00241 #     inter-node weights, and t-weights, which are terminat-to-node weights).
00242 #     
00243 #     For every region in the supplied label image a node is generated with the region-id
00244 #     as node-id. Using ndim*2-connectedness (e.g. 3*2=6 for 3D) the neighbours of each region
00245 #     are determined and a edge created for each such pair of neighbours. The provides 
00246 #     boundary_term function is used to compute the weights for all inter-node (i.e. inter
00247 #     region) weights (n-weights). For the terminal-to-node weights (t-weights) two
00248 #     approaches are taken: all regions that are touched by the supplied foreground mask
00249 #     (fg_markers) the weight from the source is set to a maximum (@link: Graph.MAX) and
00250 #     the weight to the sink to 0. For the background mask (bg_markers) vice-versa. All
00251 #     nodes that are not contained in these receive a weight calculated with the provided
00252 #     regional_term function.
00253 #     
00254 #     @param label_image: The label image as an array containing uint values. @note: The
00255 #                         region labels have to start from 1 and be continuous
00256 #                         (@link: relabel()).
00257 #     @type label_image: numpy.ndarray 
00258 #     @param fg_markers: The foreground markers as binary array.
00259 #     @type fg_markers: ndarray
00260 #     @param bg_markers: The background markers as binary array.
00261 #     @type bg_markers: ndarray
00262 #     @param regional_term: This can be either
00263 #                           False - all t-weights are set to 0, except for the nodes that are
00264 #                           directly connected to the source or sink.
00265 #                           , or a function - 
00266 #                           The supplied function is used to compute the t_edges. It has to
00267 #                           have the following signature
00268 #                           source_weight, sink_weight = fun(label_image, region_bb, region_id, fg_featurs, bg_features, regional_term_args),
00269 #                           where source_weight is the weight from the source to the node
00270 #                           of the supplied region; sink_weight the weight from the node to
00271 #                           the sink; fg_features the output of the extract_marker_features
00272 #                           function over the foreground regions if supplied, otherwise
00273 #                           False; bg_features the same for the background regions; region_bb
00274 #                           the bounding box of the current region as a list of slice-objects;
00275 #                           region_id the id og the regions; label_image the label image; and
00276 #                           regional_term_args eventual additional parameters passed via the
00277 #                           regional_term_args argument.
00278 #     @type regional_term: function
00279 #     @param boundary_term: This can be either
00280 #                           False - 
00281 #                           In which case the weight of all n_edges i.e. between all nodes
00282 #                           that are not source or sink, are set to 0.
00283 #                           , or a function -
00284 #                           In which case it is used to compute the edges weights. The
00285 #                           supplied function has to have the following signature
00286 #                           weight = fun(label_image, r1_bb, r2_bb, r1_id, r2_id, boundary_term_args),
00287 #                           where r1_bb and r2_bb are the bounding boxes around the regions
00288 #                           defined by a list of slice-objects, r1_id and r2_id the
00289 #                           respective regions ids and boundary_term_args eventual additional
00290 #                           parameters passed via the boundary_term_args argument.
00291 #                           It is called for any two neighbouring regions. If the directed
00292 #                           parameter is set to True, it is also called for the computation
00293 #                           of the reversed weight.
00294 #     @type boundary_term: function
00295 #     @param directed: Set this to True, if the n_edges should be directed i.e. have
00296 #                      different weights for both directions.
00297 #     @type directed: bool
00298 #     @param extract_marker_features: A function that has to follow the signature
00299 #                                     features = fun(label_image, bbs_of_marked_regions, is_foreground, extract_marker_features_args)
00300 #                                     and is called once with all regions that are marked
00301 #                                     by the foreground markers and once with all regions
00302 #                                     that are marked as background. The return values are
00303 #                                     passed to the regional_term function.
00304 #     @type extract_marker_features: function
00305 #     @param regional_term_args: Use this to pass some additional parameters to the
00306 #                                regional_term function.

00450 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00451 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00452 #     supplied region/label image.
00453 #     Note: The returned set contains neither duplicates, nor self-references
00454 #     (i.e. (id_1, id_1)), nor reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00455 #     
00456 #     @param label_image: An image with labeled regions (nD).
00457 #     @param return: A set with tuples denoting the edge neighbourhood.

00459 def __compute_edges(label_image):
00460     # compute the neighbours
00461     Er = __compute_edges_nd(label_image)
00462     
00463     # remove reversed neighbours and self-references
00464     for edge in list(Er):
00465         if (edge[1], edge[0]) in Er:
00466             Er.remove(edge)
00467     return Er

00471 # 
00472 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00473 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00474 #     supplied region/label image.
00475 #     Note: The returned set can contain self references (i.e. (id_1, id_1)) as well as
00476 #     reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00477 #     @see: __compute_edges_nd2 alternative implementation (slighty slower)
00478 #     @see: __compute_edges_3d faster implementation for three dimensions only
00479 #     
00480 #     @param label_image: An image with labeled regions (nD).
00481 #     @param return: A set with tuples denoting the edge neighbourhood.
00482 #     
00483 def __compute_edges_nd(label_image):
00484     Er = set()
00485     for dim in range(label_image.ndim):
00486         slices_x = []
00487         slices_y = []
00488         for di in range(label_image.ndim):
00489             slices_x.append(slice(None, -1 if di == dim else None))
00490             slices_y.append(slice(1 if di == dim else None, None))
00491         Er = Er.union(set(zip(label_image[slices_x].flat,
00492                               label_image[slices_y].flat)))
00493     return Er

00496 # 
00497 #     Computes the region neighbourhood defined by a star shaped n-dimensional structuring
00498 #     element (as returned by scipy.ndimage.generate_binary_structure(ndim, 1)) for the
00499 #     supplied region/label image.
00500 #     Note: The returned set can contain self references (i.e. (id_1, id_1)) as well as
00501 #     reversed references (e.g. (id_1, id_2) and (id_2, id_1).
00502 #     @see: __compute_edges_nd alternative implementation (slightly faster)
00503 #     @see: __compute_edges_3d faster implementation for three dimensions only
00504 #     
00505 #     @param label_image: An image with labeled regions (nD).
00506 #     @param return: A set with tuples denoting the edge neighbourhood.
00507 #     
00508 def __compute_edges_nd2(label_image):
00509     # prepare iterators
00510     labels = label_image.flat
00511     nb_pointers = []
00512     for dim in range(label_image.ndim):
00513         slices_base = []
00514         slices_add = []
00515         for di in range(label_image.ndim):
00516             slices_base.append(slice(1 if di == dim else None, None))
00517             slices_add.append(slice(-1 if di == dim else None, None))
00518         nb_pointers.append(scipy.concatenate((label_image[slices_base],
00519                                               label_image[slices_add]), dim).flat)
00520     
00521     # iterate and search neighbours
00522     Er = set()
00523     for label, nbs in zip(labels, zip(*nb_pointers)):
00524         for nb in nbs: Er.add((label, nb))
00525     return Er

00609     Er = Er.union(__compute_edges3d_1d(label_image[:,-1,-1]))
