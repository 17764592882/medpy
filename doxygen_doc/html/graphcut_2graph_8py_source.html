00025 #     Represents a graph suitable for further processing with the graphcut package.
00026 #     
00027 #     The graph contains nodes, edges (directed) between the nodes (n-edges), edges
00028 #     between two terminals (called source and sink) and the nodes (t-edges), and a
00029 #     weight for each edge. 
00030 #     
00031 #     @note the node-ids used by the graph are assumed to start with 1 and be
00032 #     continuous. This is not actually checked, except when calling the
00033 #     inconsistent() method, so be careful.
00034 #     
00035 #     Methods:
00036 #         - set_nodes() -- Set the nodes.

00038 #         - set_sink_nodes() -- Set the sink nodes.
00039 #         - set_nweights() -- Set the inter-node weights-
00040 #         - add_tweights() -- Add some terminal-to-node weights.
00041 #         - get_node_count() -- Return the number of nodes
00042 #         - get_nodes() -- Return all nodes.
00043 #         - get_source_nodes() -- Return all nodes connected to the source.
00044 #         - get_sink_nodes() -- Return all nodes connected to the sink.
00045 #         - get_edges() -- Return all edges.

00047 #         - get_tweights() -- Return all terminal-node weights.

00202     #         Perform some consistency tests on the graph represented by this object.
00203     #         @note this check is very time intensive and should not be executed on huge
00204     #         graphs, except for debugging purposes.
00205     #         @return False if consistent, else a list of inconsistency messages.

00207     def inconsistent(self):
00208         messages = []
00209         for node in self.__tweights.iterkeys():
00210             if not node <= self.__nodes: messages.append("Node {} in t-weights but not in nodes.".format(node))
00211         for node in self.__snodes:
00212             if not node <= self.__nodes: messages.append("Node {} in s-nodes but not in nodes.".format(node))
00213         for node in self.__tnodes:
00214             if not node <= self.__nodes: messages.append("Node {} in t-nodes but not in nodes.".format(node))
00215         for e in self.__nweights.iterkeys():
00216             if not e[0] <= self.__nodes: messages.append("Node {} in edge {} but not in nodes.".format(e[0], e))
00217             if not e[1] <= self.__nodes: messages.append("Node {} in edge {} but not in nodes.".format(e[1], e))
00218             if (e[1], e[0]) in self.__nweights.iterkeys(): messages.append("The reversed edges of {} is also in the n-weights.".format(e))
00219                 
00220             
00221         if 0 == len(messages): return False
00222         else: return messages

00343     def set_nweight(self, node_from, node_to, weight_there, weight_back):
00344         if node_from >= self.__nodes or node_from < 0:
00345             raise ValueError('Invalid node id (node_from) of {}. Valid values are 0 to {}.'.format(node_from, self.__nodes - 1))
00346         elif node_to >= self.__nodes or node_to < 0:
00347             raise ValueError('Invalid node id (node_to) of {}. Valid values are 0 to {}.'.format(node_to, self.__nodes - 1))
00348         elif node_from == node_to:
00349             raise ValueError('The node_from ({}) can not be equal to the node_to ({}) (self-connections are forbidden in graph cuts).'.format(node_from, node_to))
00350         elif weight_there <= 0 or weight_back <= 0:
