00030 __description__ = """
00031                   Perform post-morphology operations over a gradient image as required
00032                   for a viscous watershed.
00033                   This step is a pre-processing step that has to be followed by a
00034                   standard watershed segmentation, but makes it behave as if the
00035                   a viscous liquid was used.
00036                   The gradient images histogram is divided into bins roughly containing
00037                   the same number of voxels and on each a closing operation is performed.
00038                   The number of bin and the size of the sphere used for the closing over
00039                   each is determined by the provided discrete function.
00040                   The resulting image will be saved under the same name and type as the
00041                   input image, only with an '_wviscous' and its parameters as suffix.
00042                   See "The viscous watershed transform" by Vachier, Corinne and Meyer,
00043                   Fernand (Journal of Mathematical Imaging and Vision) for more details.
00044                   """

00118             
00119             # add result of this slice to the general results

00154 def hist_flatened(im,nbr_bins=10):
00155     #get image histogram
00156     imhist, bins = histogram(im.flatten(), 1000)
00157     
00158     # only take bins with content into account
00159     nz = imhist.nonzero()
00160     imhist = imhist[nz]
00161     bins = bins[nz]
00162     
00163     # prepare iteration
00164     bins_final = [bins[0]] # set initial bin delimiter
00165     bins_content = scipy.prod(im.shape) / float(nbr_bins)
00166     tmp_content = 0
00167     for i in range(len(imhist) - 1):
00168         tmp_content += imhist[i]
00169         if tmp_content >= bins_content: # bin full
00170             #bins_final.append(bins[i+1]) # add new bin delimiter
00171             #tmp_content = 0
00172             div = float(imhist[i]) / (bins_content - (tmp_content - imhist[i])) # what i got / what i want
00173             bins_final.append(bins[i] + (bins[i+1] - bins[i]) / div) # append a partial bin border, assuming that the dist inside the bin in equal
00174             tmp_content = imhist[i] - (bins_content - (tmp_content - imhist[i]))
00175             
00176     bins_final.append(im.max() + 1) # one added to work against rounding errors
