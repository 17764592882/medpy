00076 # 
00077 #     Generate a C++ file executing the Boyov and Kolmogorovs max-flow/min-cut algorithm
00078 #     over the supplied graph.
00079 #     
00080 #     @param graphs: The graph representing the problem. 
00081 #     @type graphs: medpy.graphcut.Graph
00082 #     @return: A compile-ready, pre-formated version of the problem to be compiled/executed
00083 #              either using the methods provided by the @link: cut module or manually.
00084 #     @rtype: str
00085 #     
00086 #     @raise SubprocessError: When the algorithm library can not be accessed.
00087 #     
00088 def bk_mfmc_generate(graph):
00089     # prepare logger
00090     logger = Logger.getInstance()
00091     
00092     logger.info('Checking conditions for BK_MFMC source file creation...')
00093     
00094     # get the algorithms library
00095     library = __bk_mfmc_get_library()
00096     
00097     # format header and footer
00098     header = __BK_MFMC_HEADER.format(library['graph_lib'], len(graph.get_tweights()), len(graph.get_nweights()))
00099     footer = __BK_MFMC_FOOTER.format(len(graph.get_nodes()), __BK_MFMC_SOURCE_MARKER, __BK_MFMC_SINK_MARKER)
00100     
00101     logger.info('Generating BK_MFMC C++ source file...')
00102     
00103     # generate cpp file with the graph encoded
00104     # Note: The min-cut/max-flow algorithm required node ids to start from 0, while the
00105     # regions ids and therefore node ids returned by the graph object start from one.
00106     # The ids written to the file are therefore decremented by 1. This is taken into
00107     # account in the way the result is printed (i.e. increment of 1, see the __FOOTER)
00108     source_file = header
00109     source_file += '\tg -> add_node({});\n\n'.format(len(graph.get_nodes()))
00110     for node_id in graph.get_tweights():
00111         source_file += '\tg -> add_tweights({}, {}, {});\n'.format(node_id - 1, graph.get_tweights()[node_id][0], graph.get_tweights()[node_id][1])
00112     source_file += '\n'
00113     for edge in graph.get_nweights():
00114         source_file += '\tg -> add_edge({}, {}, {}, {});\n'.format(edge[0] - 1, edge[1] - 1, graph.get_nweights()[edge], graph.get_nweights()[edge])
00115     source_file += '\n'
00116     source_file += footer
00117     
00118     return source_file

00126 def __bk_mfmc_get_library():
00127     library = {}
00128     
00129     # access the algorithms location
00130     try:
00131         library['graph_lib'] = medpy.__path__[0] + __BK_MFMC_GRAPH_LIB
00132         library['graph_file'] = medpy.__path__[0] + __BK_MFMC_GRAPH_FILE
00133         library['maxflow_file'] = medpy.__path__[0] + __BK_MFMC_MAXFLOW_FILE
00134     except Exception as e:
00135         raise SubprocessError('The path to the algorithm could not be accessed, reason: {}.'.format(e.message))
00136     
00137     # test if the algorithm files exist
00138     if not (os.path.exists(library['graph_lib']) or
00139             os.path.exists(library['graph_file']) or 
00140             os.path.exists(library['maxflow_file'])):
00141         raise SubprocessError('At least one of the required algorithm files does not seem to exists ({}, {}, {}).'.format(library['graph_lib'], library['graph_file'], library['maxflow_file']))
00142
