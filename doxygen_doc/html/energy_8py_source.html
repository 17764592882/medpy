00032 # 
00033 #     An implementation of the boundary term in (1), suitable to be used with the
00034 #     @link: graph_from_labels() function.
00035 #     
00036 #     Determines for each two supplied regions the voxels forming their border assuming
00037 #     ndim*2-connectedness (e.g. 3*2=6 for 3D). From the gradient magnitude values of each
00038 #     end-point voxel the border-voxel pairs, the highest one is selected and passed to a
00039 #     strictly positive and decreasing function g, which is defined as:
00040 #     \f[
00041 #         g(x) = \left(\frac{1}{1+|x|}\right)^k
00042 #     \f]
00043 #     ,where \f$k=2\f$. The final weight \f$w_{i,j}\f$ between two regions \f$r_i\f$ and
00044 #     \f$r_j\f$ is then determined by the sum of all these neighbour values:
00045 #     \f[
00046 #         w = \sum_{e_{m,n}|inF_{(r_i,r_j)}}g(\max(|I(m)|,|I(n)|))
00047 #     \f]
00048 #     , where \f$F_{(r_i,r_j)}\f$ is the set of border voxel-pairs \f$e_{m,n}\f$ between
00049 #     the regions \f$r_i\f$ and \f$r_j\f$ and \f$|I(p)|\f$ the absolute of the gradient
00050 #     magnitude at the voxel \f$p\f$
00051 #     
00052 #     This boundary_function works as an edge indicator in the original image. In simpler
00053 #     words the weight (and therefore the energy) is obtained by summing the local contrast
00054 #     along the boundaries between two regions.
00055 #     
00056 #     @note This function requires the gradient magnitude image of the original image to
00057 #     be passed along. That means that @link: graph_from_labels() has to be called with
00058 #     boundary_term_args set to the gradient image. 
00059 #     
00060 #     @param label_image: the label image
00061 #     @type label_image: numpy.ndarray
00062 #     @param r1_bb: The bounding box of the first region, as a list of slice objects.
00063 #     @type r1_bb: list
00064 #     @param r2_bb: The bounding box of the second region, as a list of slice objects.
00065 #     @type r2_bb: list
00066 #     @param r1_id: The region/node id of the first region.
00067 #     @type r1_id: int
00068 #     @param r2_id: The region/node id of the second region.
00069 #     @type r2_id: int
00070 #     @param gradient_image: The gradient magnitude image of the original image.
00071 #     @type gradient_image: numpy.ndarray
00072 #     
00073 def boundary_stawiaski(label_image, r1_bb, r2_bb, r1_id, r2_id, (gradient_image)):
00074     # compute the region of interest (the intersection of the bounding boxes)
00075     roi = points_to_slices(intersection(slices_to_points(r1_bb),
00076                                         slices_to_points(r2_bb)))
00077     
00078     # prepare strictly positive and decreasing function (Note: input is expected to be a tuple)
00079     g = lambda x: math.pow(1./(1. + max(map(abs, x))), 2) # k = 2
00080     
00081     # prepare iterator lists (parallel iterator over all possible neighbourhood voxels and corresponding value pairs combinations)
00082     values, labels = __boundary_stawiaski_get_nb_iterators(gradient_image[roi], label_image[roi])
00083     
00084     # iterate over values and compute results
00085     sumand = 0
00086     for label, value in zip(labels, values):
00087         if (r1_id, r2_id) == label:
00088             sumand += g(value)
00089     return sumand

00092 # Returns iterators over a ndim*2-connectedness neighbourhood parallel over both supplied images.
00093 def __boundary_stawiaski_get_nb_iterators(image, label_image):
00094     values = []
00095     labels = []
00096     for dim in range(image.ndim):
00097         slices_x = []
00098         slices_y = []
00099         for di in range(image.ndim):
00100             slices_x.append(slice(None, -1 if di == dim else None))
00101             slices_y.append(slice(1 if di == dim else None, None))
00102         values.extend(zip(image[slices_x].flat,
00103                           image[slices_y].flat))
00104         labels.extend(zip(label_image[slices_x].flat,
00105                           label_image[slices_y].flat))
00106     return (values, labels)

00118     rect = ([],[]) # lower-left / upper-right
00119     for i in range(len(slices)):
00120         rect[0].append(max(slices[i].start - 1, 0))
00121         rect[1].append(slices[i].stop + 1)
00122         rect[0].append(max(slices[i].start - 1, 0))
00123         rect[1].append(slices[i].stop + 1)

00145 #     Computes the intersection rectangle of two nD rectangles increased.
00146 #     The intersection window marks the area in which the two images bounding boxes
00147 #     intersect. The returned rectangle is defined by its lower-left and upper-right
00148 #     corner.
00149 #     
00150 #     @param rect1: The first rectangle defined by its lower-left and upper-right
00151 #                   corners (lower_left, uper_right).
00152 #     @param rect2: The second rectangle defined by its lower-left and upper-right
00153 #                   corners (lower_left, uper_right).
00154 #     @return: The intersection rectangle defined by a sequence of slice objects or False
00155 #              if no intersection occured.
00156 #     
00157 #     
00158 def intersection(rect1, rect2): # slightly tested
00159     # check if an intersection occurs at all
00160     intersect = rect1[0][0] < rect2[1][0] and \
00161                 rect1[0][1] < rect2[1][1] and \
00162                 rect1[0][2] < rect2[1][2] and \
00163                 rect1[1][0] > rect2[0][0] and \
00164                 rect1[1][1] > rect2[0][1] and \
00165                 rect1[1][2] > rect2[0][2]    
00166     
00167     # if not, return False
00168     if not intersect:
00169         return False
00170     
00171     # otherwise compute intersection cube
00172     intersection = ((max(rect1[0][0], rect2[0][0]), #x lower
00173                      max(rect1[0][1], rect2[0][1]), #y lower
00174                      max(rect1[0][2], rect2[0][2])), #z lower
00175                     (min(rect1[1][0], rect2[1][0]), #x upper
00176                      min(rect1[1][1], rect2[1][1]), #y upper
00177                      min(rect1[1][2], rect2[1][2]))) #z upper
00178     
00179     return intersection
