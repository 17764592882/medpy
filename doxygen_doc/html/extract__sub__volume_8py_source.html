00030 __description__ = """
00031                   Takes a medical image of arbitrary dimensions and the dimensions
00032                   of a sub-volume that lies inside the dimensions of this images.
00033                   Extracts the sub-volume from the supplied image and saves it.
00034                   
00035                   The volume to be extracted is defined by its slices, the syntax is the same as
00036                   for numpy array indexes (i.e. starting with zero-index, the first literal (x) of any
00037                   x:y included and the second (y) excluded).
00038                   E.g. '2:3,4:6' would extract the slice no. 3 in X and 5, 6 in Y direction of a 2D image.
00039                   E.g. '99:199,149:199,99:249' would extract the respective slices in X,Y and Z direction of a 3D image.
00040                        This could, for example, be used to extract the area of the liver form a CT scan.
00041                   To keep all slices in one direction just omit the respective value:
00042                   E.g. '99:199,149:199,' would work ust as example II, but extract all Z slices.
00043                        Note here the trailing colon.
00044                        
00045                   Note that the volume dimensions must be the same as the image dimensions, otherwise as exception is thrown.
00046                   Note that the input images offset is not taken into account except for Nifti images.
00047                   Note that the sub-volume is not checked to actually be inside the image dimensions.
00048                   Note to take into account the input images orientation when supplying the sub-volume.
00049                   """

00086     index = [slice(x[0], x[1]) for x in args.volume]

00090     if 0 == len(volume):

00096     # get base origin of the image
