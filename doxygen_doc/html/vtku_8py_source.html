00031 def getInformation(image): #tested
00032     assert isinstance(image, vtk.vtkImageData)

00034     # refresh information

00038     # request information and format string

00042     s += '\tdimensions: ' + str(image.GetDimensions()) + '\n'

00044     s += '\tspacing: ' + str(image.GetSpacing()) + '\n'
